\usepackage{tikz}

\tikzstyle{annot}=[font=\fontsize{17}{0}\selectfont,text width=7em, text centered]
\tikzstyle{captionLabel}=[annot,font=\fontsize{20}{20}\selectfont,thick,text width=200]

\newcommand{\threeDimMatrix}[4]{% width, height, depth, scale
  \begin{tikzpicture}[scale=#4, line width=0.15em]
    \foreach \x in {0,...,#1}{
      \draw (\x ,0  ,#3 ) -- (\x ,#2 ,#3 );
      \draw (\x ,#2 ,#3 ) -- (\x ,#2 ,0  );
    }

    \foreach \x in {0,...,#2}{
      \draw (#1 ,\x ,#3 ) -- (#1 ,\x ,0  );
      \draw (0  ,\x ,#3 ) -- (#1 ,\x ,#3 );
    }

    \foreach \x in {0,...,#3}{
      \draw (#1 ,0  ,\x ) -- (#1 ,#2 ,\x );
      \draw (0  ,#2 ,\x ) -- (#1 ,#2 ,\x );
    }

    \node[annot] at (1.5, -1.5) {axis 0};
    \node[annot] at (-2, 1) {axis 1};
    \node[annot] at (-1.25, 3.65) {axis 2};

    \node[captionLabel] at (1.5, -2.5) {\bfseries 3D array};
  \end{tikzpicture}
}
